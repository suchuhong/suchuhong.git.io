贝叶斯滤波

$$
\text { 重要定理：若 }f_{X}(x) \text { ~ } N\left(\mu_{1}, \sigma_{1}^{2}\right), \quad f_{Y |X}(y | x) \text { ~ } N\left(\mu_{2}, \sigma_{2}^{2}\right)
$$

$$
\text { 则 }f_{X |Y}(x | y)=\left(\frac{\sigma_{1}^{2}}{\sigma_{1}^{2}+\sigma_{2}^{2}} \mu_{2}+\frac{\sigma_{2}^{2}}{\sigma_{1}^{2}+\sigma_{2}^{2}} \mu_{1}, \frac{\sigma_{1}^{2} \sigma_{2}^{2}}{\sigma_{1}^{2}+\sigma_{2}^{2}}\right)
$$

$$
\text { 若 } \quad \sigma_{1}^{2}\gg\sigma_{2}^{2} \text { 后验 ~} N\left(\frac{1}{1+\left(\sigma_{2}^{2} / \sigma_{1}^{2}\right)} \mu_{2}+\frac{\sigma_{2}^{2} / \sigma_{1}^{2}}{1+\left(\sigma_{2}^{2} / \sigma_{1}^{2}\right)} \mu_{1}, \frac{\sigma_{2}^{2}}{1+\left(\sigma_{2}^{2} / \sigma_{1}^{2}\right)}\right) \text { ~ } N\left(\mu_{2}, \sigma_{2}^{2}\right)
$$

$$
\text { 若 } \quad \sigma_{1}^{2}\ll\sigma_{2}^{2} \text { 后验 ~ } N\left(\frac{\sigma_{1}^{2} / \sigma_{2}^{2}}{1+\left(\sigma_{1}^{2} / \sigma_{2}^{2}\right)} \mu_{2}+\frac{1}{1+\left(\sigma_{1}^{2} / \sigma_{2}^{2}\right)} \mu_{1}, \frac{\sigma_{1}^{2}}{1+\left(\sigma_{1}^{2} / \sigma_{2}^{2}\right)}\right) \text { ~ } N\left(\mu_{1}, \sigma_{1}^{2}\right)
$$

$$
\begin{equation}\sigma_{1}^{2} \gg \sigma_{2}^{2} \quad \text { 倾向于观测值 } \sigma_{2}^{2} \ll \sigma_{1}^{2} \quad \text { 倾向于预测值 }\end{equation}
$$

$$
\frac{\sigma_{1}^{2} \sigma_{2}^{2}}{\sigma_{1}^{2}+\sigma_{2}^{2}}=\sigma_{1}^{2} \frac{1}{1+\sigma_{1}^{2} / \sigma_{2}^{2}}=\sigma_{2}^{2} \frac{1}{1+\sigma_{2}^{2} / \sigma_{1}^{2}}
$$

贝叶斯公式

$$
P(X=x ，Y=y)=P(Y = y | X=x) P(X=x)=P(X=x | Y = y) P(Y = y)
$$

1、离散形式

$$
P(X=x | Y=y)=\frac{P(Y = y | X=x) P(X=x)}{P(Y=y)}
$$

2、连续形式

$$
P(X<x | Y=y)=\frac{P(Y=y | X<x) P(X<x)}{P(Y=y)}
